### 7.3 샘플 테스트 (GeminiEngine)

```python
# tests/unit/engines/test_gemini_engine.py
import pytest
from unittest.mock import MagicMock, patch
from PIL import Image
from engines.gemini_engine import GeminiEngine


class TestGeminiEngine:
    def test_supports_image_and_ocr_flags(self):
        engine = GeminiEngine(api_key="fake", model="gemini-2.0-flash", tracker=MagicMock())
        assert engine.supports_image is True
        assert engine.supports_ocr is False

    def test_extract_table_success(self, load_mock, mocker):
        # Why mock 경로:
        #   GeminiEngine.__init__ 이 `import google.generativeai as genai` 를
        #   지연 import 하여 self._genai 에 저장한 뒤, extract_table() 에서
        #   self._genai.GenerativeModel(...).generate_content(...) 로 체인 호출한다.
        #   따라서 모듈 레벨 alias("engines.gemini_engine.genai") 를 patch 해야
        #   __init__ 이후의 모든 호출이 Mock 으로 치환된다.
        mock_response = MagicMock()
        mock_response.text = load_mock("gemini/extract_table_success.html")
        mock_response.usage_metadata = MagicMock(
            prompt_token_count=100, candidates_token_count=200
        )
        mock_model = MagicMock()
        mock_model.generate_content.return_value = mock_response
        mock_genai = MagicMock()
        mock_genai.GenerativeModel.return_value = mock_model
        mocker.patch("engines.gemini_engine.genai", mock_genai)

        engine = GeminiEngine(api_key="fake", model="gemini-2.0-flash", tracker=MagicMock())
        # __init__ 직후 self._genai 가 실제 패키지로 바뀌므로 Mock 으로 재주입
        engine._genai = mock_genai

        img = Image.new("RGB", (100, 100))
        html, in_tok, out_tok = engine.extract_table(img, table_num=1)

        assert "<table" in html
        assert in_tok == 100 and out_tok == 200

    def test_extract_table_api_error_returns_safe_fallback(self, mocker):
        mock_model = MagicMock()
        mock_model.generate_content.side_effect = Exception("quota exceeded")
        mock_genai = MagicMock()
        mock_genai.GenerativeModel.return_value = mock_model
        mocker.patch("engines.gemini_engine.genai", mock_genai)

        engine = GeminiEngine(api_key="fake", model="gemini-2.0-flash", tracker=MagicMock())
        engine._genai = mock_genai

        img = Image.new("RGB", (100, 100))
        html, in_tok, out_tok = engine.extract_table(img, table_num=1)

        assert "<!--" in html  # 에러 주석 포함
        assert in_tok == 0 and out_tok == 0

    def test_cache_hit_skips_api_call(self, mocker):
        cache_mock = MagicMock()
        cache_mock.get.return_value = "<table>cached</table>"
        mock_model = MagicMock()
        mock_genai = MagicMock()
        mock_genai.GenerativeModel.return_value = mock_model
        mocker.patch("engines.gemini_engine.genai", mock_genai)

        engine = GeminiEngine(api_key="fake", model="gemini-2.0-flash", tracker=MagicMock())
        engine._genai = mock_genai
        engine.cache = cache_mock

        img = Image.new("RGB", (100, 100))
        html, _, _ = engine.extract_table(img, table_num=1)

        assert "cached" in html
        mock_model.generate_content.assert_not_called()
```

```python
# tests/unit/exporters/test_excel_exporter.py
import pytest
from exporters.excel_exporter import _classify_table, _try_parse_number


class TestClassifyTable:
    # Note: 반환값은 한글이 아닌 영문 코드("estimate"/"detail"/"condition"/"generic").
    #       Excel 시트명은 export() 레이어에서 이 코드를 한글로 매핑한다.
    def test_estimate_by_header(self):
        table = {"headers": ["명 칭", "규 격", "수량", "금 액"], "rows": []}
        assert _classify_table(table) == "estimate"

    def test_detail_by_header(self):
        table = {"headers": ["품명", "수량", "합계_금액"], "rows": []}
        assert _classify_table(table) == "detail"

    def test_condition_by_keyword(self):
        table = {"headers": ["일반사항"], "rows": []}
        assert _classify_table(table) == "condition"

    def test_generic_fallback(self):
        table = {"headers": ["X", "Y"], "rows": []}
        assert _classify_table(table) == "generic"


class TestParseNumber:
    # 현재 스펙(`_RE_NUMERIC = r'^-?[\d,]+\.?\d*$'`)에서 허용되는 입력:
    #   - 정수/소수, 선행 `-`, 콤마 천단위 구분자만
    # 미지원(모두 None 반환):
    #   - 괄호 음수 표기 `(5,000)`
    #   - 한글/영문 단위 suffix `"1,234원"`, `"100KW"`
    #   - 선행 0 식별자 `"0015"` (코드 보호)
    # 이 테스트는 **현재 구현 스펙**을 고정하기 위한 기준선이며,
    # 괄호/단위 파싱이 추후 추가되면 이 테스트를 먼저 수정한 뒤 구현을 확장한다.
    @pytest.mark.parametrize("text,expected", [
        ("1,000,000", 1_000_000),   # 정수 (콤마 제거)
        ("  3.5  ", 3.5),            # 공백 strip + 소수
        ("-500", -500),              # 선행 음수
        ("1,234원", None),           # 단위 suffix 미지원 → None
        ("(5,000)", None),           # 괄호 음수 미지원 → None
        ("0015", None),              # 선행 0 식별자 보호 → None
        ("-", None),                 # 대시 단독 → None
        ("", None),                  # 빈 문자열 → None
        ("N/A", None),               # 비숫자 → None
    ])
    def test_parse_variants(self, text, expected):
        assert _try_parse_number(text) == expected
```
