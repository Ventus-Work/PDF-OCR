```python
import glob, shutil, platform

def _detect_poppler_path() -> str | None:
    # 1. 환경변수
    if env := os.environ.get("POPPLER_PATH"):
        if Path(env).exists(): return env
    # 2. PATH (`pdftotext` 위치)
    if w := shutil.which("pdftotext"):
        return str(Path(w).parent)
    # 3. OS별 기본
    if platform.system() == "Windows":
        hits = glob.glob(r"C:\poppler\poppler-*\Library\bin")
        if hits: return sorted(hits)[-1]  # 최신
    elif platform.system() == "Darwin":
        for p in ["/opt/homebrew/bin", "/usr/local/bin"]:
            if (Path(p) / "pdftotext").exists(): return p
    else:  # Linux
        return "/usr/bin" if Path("/usr/bin/pdftotext").exists() else None
    return None
```

```python
def get_extractor(path: Path):
    suffix = path.suffix.lower()
    mapping = {
        ".pdf": "hybrid",
        ".hwpx": "hwpx",
        ".docx": "docx",
        ".md": None,  # Phase 1 건너뜀
    }
    key = mapping.get(suffix)
    if key is None and suffix != ".md":
        raise ParserError(f"지원하지 않는 형식: {suffix}")
    return key
```
